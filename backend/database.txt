CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    role VARCHAR(20) DEFAULT 'patient'
);

CREATE TABLE dentists (
    id SERIAL PRIMARY KEY,
    first_name VARCHAR(50) NOT NULL,
    last_name VARCHAR(50) NOT NULL,
    specialization VARCHAR(100),
    email VARCHAR(100),
    phone_number VARCHAR(20)
);

CREATE TABLE services (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    price DECIMAL(10, 2) NOT NULL
);

-- TABELA POŚREDNICZĄCA: Przypisuje zabiegi do lekarzy

CREATE TABLE dentist_services (
    dentist_id INTEGER REFERENCES dentists(id) ON DELETE CASCADE,
    service_id INTEGER REFERENCES services(id) ON DELETE CASCADE,
    PRIMARY KEY (dentist_id, service_id)
);


CREATE TABLE available_slots (
    id SERIAL PRIMARY KEY,
    dentist_id INTEGER REFERENCES dentists(id) ON DELETE CASCADE,
    slot_date DATE NOT NULL,
    start_time TIME NOT NULL,
    end_time TIME NOT NULL,
    is_available BOOLEAN DEFAULT TRUE,
    service_name VARCHAR(100) DEFAULT 'Konsultacja'
);

CREATE TABLE appointments (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    dentist_id INTEGER REFERENCES dentists(id) ON DELETE CASCADE,
    slot_id INTEGER UNIQUE REFERENCES available_slots(id) ON DELETE CASCADE, -- <--- DODANE SŁOWO 'UNIQUE'
    status VARCHAR(20) DEFAULT 'booked',
    custom_service VARCHAR(100),
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

CREATE TABLE notifications (
    id SERIAL PRIMARY KEY,
    user_id INTEGER REFERENCES users(id) ON DELETE CASCADE,
    message TEXT NOT NULL,
    is_read BOOLEAN DEFAULT FALSE,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

INSERT INTO available_slots (dentist_id, slot_date, start_time, end_time, is_available, service_name)
SELECT
    d.id,
    CURRENT_DATE + i,
    ('09:00:00'::time + (j * INTERVAL '1 hour'))::time,
    ('09:30:00'::time + (j * INTERVAL '1 hour'))::time,
    TRUE,
    CASE
        WHEN d.specialization = 'stomatologia zachowawcza' THEN 'Przegląd stomatologiczny'
        WHEN d.specialization = 'endodoncja' THEN 'Leczenie kanałowe jednokanałowe'
        WHEN d.specialization = 'ortodoncja' THEN 'Konsultacja ortodontyczna'
        WHEN d.specialization = 'chirurgia stomatologiczna' THEN 'Ekstrakcja zęba'
        WHEN d.specialization = 'protetyka' THEN 'Korona porcelanowa'
        WHEN d.specialization = 'implantologia' THEN 'Implant stomatologiczny (wszczepienie)'
        WHEN d.specialization = 'periodontologia' THEN 'Leczenie zapalenia dziąseł'
        WHEN d.specialization = 'stomatologia dziecięca' THEN 'Wizyta adaptacyjna dziecka'
        ELSE 'Konsultacja'
    END
FROM dentists d
CROSS JOIN generate_series(1, 4) AS i
CROSS JOIN generate_series(0, 4) AS j;
